Idle: Flota suavemente, parpadea, mira alrededor.

Sleeping: Los ojos se apagan, la cabeza cae, aparecen las "Zzz".

let busy = false;
const history = [];

async function sendMsg(){
  const inp = document.getElementById('mainInp');
  const text = inp.value.trim();
  if(!text || busy) return;
  inp.value='';
  busy=true;
  history.push({role:'user',content:text});
  setRobotState('thinking');
  document.getElementById('msgBubble').classList.add('glow');
  document.getElementById('msgText').textContent='...';
  document.getElementById('msgText').classList.add('msg-typing');

  try{
    const res = await fetch('https://api.anthropic.com/v1/messages',{
      method:'POST',
      headers:{'Content-Type':'application/json'},
      body:JSON.stringify({
        model:'claude-sonnet-4-20250514',max_tokens:1000,
        system:`Eres Geo, asistente personal inteligente de un OS llamado Geo OS. Eres como Jarvis: proactivo, conciso, personal. Conoces los proyectos del usuario: Sitio Web (65%), Transformación Física (28%), Geo Bot producto (12%). Responde en máximo 2 frases. Idioma: español.`,
        messages:history
      })
    });
    const d = await res.json();
    const reply = d.content?.[0]?.text || 'Entendido.';
    history.push({role:'assistant',content:reply});
    respondMsg(reply);
  } catch(e){
    respondMsg('Error de conexión. Intenta de nuevo.');
  }
  busy=false;
}

function respondMsg(text){
  const el = document.getElementById('msgText');
  el.classList.remove('msg-typing');
  setRobotState('speaking');
  let i=0;
  el.textContent='';
  el.classList.add('msg-typing');
  const spd = Math.max(14, Math.min(28, 1000/text.length));
  const iv = setInterval(()=>{
    el.textContent += text[i++];
    if(i>=text.length){
      clearInterval(iv);
      el.classList.remove('msg-typing');
      setTimeout(()=>setRobotState('idle'),1200);
    }
  },spd);
}

    else if(i > ti) el.classList.add('right');
    else el.classList.add('active');
  });

  if(s==='avatar') document.getElementById('nav-home').classList.add('active');
  if(s==='vida')   document.getElementById('nav-vida').classList.add('active');

  // pip avatar
  const pip = document.getElementById('pipAvatar');
  pip.classList.toggle('show', s!=='avatar');

  currentScreen = s;
}

let busy = false;
const history = [];

async function sendMsg(){
  const inp = document.getElementById('mainInp');
  const text = inp.value.trim();
  if(!text || busy) return;
  inp.value='';
  busy=true;
  history.push({role:'user',content:text});
  setRobotState('thinking');
  document.getElementById('msgBubble').classList.add('glow');
  document.getElementById('msgText').textContent='...';
  document.getElementById('msgText').classList.add('msg-typing');

  try{
    const res = await fetch('https://api.anthropic.com/v1/messages',{
      method:'POST',
      headers:{'Content-Type':'application/json'},
      body:JSON.stringify({
        model:'claude-sonnet-4-20250514',max_tokens:1000,
        system:`Eres Geo, asistente personal inteligente de un OS llamado Geo OS. Eres como Jarvis: proactivo, conciso, personal. Conoces los proyectos del usuario: Sitio Web (65%), Transformación Física (28%), Geo Bot producto (12%). Responde en máximo 2 frases. Idioma: español.`,
        messages:history
      })
    });
    const d = await res.json();
    const reply = d.content?.[0]?.text || 'Entendido.';
    history.push({role:'assistant',content:reply});
    respondMsg(reply);
  } catch(e){
    respondMsg('Error de conexión. Intenta de nuevo.');
  }
  busy=false;
}

function respondMsg(text){
  const el = document.getElementById('msgText');
  el.classList.remove('msg-typing');
  setRobotState('speaking');
  let i=0;
  el.textContent='';
  el.classList.add('msg-typing');
  const spd = Math.max(14, Math.min(28, 1000/text.length));
  const iv = setInterval(()=>{
    el.textContent += text[i++];
    if(i>=text.length){
      clearInterval(iv);
      el.classList.remove('msg-typing');
      setTimeout(()=>setRobotState('idle'),1200);
    }
  },spd);
}
